### 4) Come sono definite le pratiche selvicolturali?

SELECT ?comma ?explanandum ?explanans ?function ?reference
WHERE {
{
?explanandum t3:actor t3:2c_si_definisce_come .
?explanandum t3:hasReference ?comma .
}
UNION
{
?explanans t3:undergoer t3:2c_si_definisce_come .
?explanans t3:hasFunction ?function .
FILTER NOT EXISTS {
?function t3:hasMetaAttr t3:di_quanto_previsto_alla_lettera_d .
}
}
UNION
{
?explanans t3:undergoer t3:2c_si_definisce_come .
?explanans t3:hasFunction ?function .
?function t3:hasMetaAttr ?reference .
}
}
