### 11) Come sono definiti i terreni silenti?

SELECT ?comma ?explanandum ?explanans ?reference ?condition
WHERE {
{
?explanandum t3:actor t3:2h_sono_definiti_come .
?explanandum t3:hasReference ?comma .
}
UNION
{
?explanans t3:undergoer t3:2h_sono_definiti_come .
?explanans t3:hasReference ?reference .
?explanans t3:hasCondition ?condition .
}
}
