### 22) Come viene definito il bosco?

SELECT ?comma ?explanandum ?explanans ?condition ?metric
WHERE {
{
?explanandum t3:actor t3:3_si_definisce_come .
?explanandum t3:hasReference ?comma .
}
UNION
{
?explanans t3:undergoer t3:3_si_definisce_come .
?explanans t3:hasCondition ?condition .
}
UNION
{
?explanans t3:undergoer t3:3_si_definisce_come .
?explanans t3:hasMetric ?metric .
}
}
