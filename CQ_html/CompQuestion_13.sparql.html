### 13) Come sono definite le superfici usate per il pascolo del bestiame?

SELECT ?comma ?explanandum ?explanans ?function ?condition
WHERE {
{
?explanandum t3:actor t3:2i2_sono_definite_come .
?explanandum t3:hasReference ?comma .
}
UNION
{
?explanans t3:undergoer t3:2i2_sono_definite_come .
?explanans t3:hasFunction ?function .
?function t3:hasMetaAttr ?condition .
}
}
