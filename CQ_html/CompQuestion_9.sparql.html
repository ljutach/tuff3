### 9) Come vengono definiti i terreni abbandonati caratterizzati dal bosco d'alto fusto?

SELECT ?comma ?explanandum ?explanans ?condition
WHERE {
{
?explanandum t3:actor t3:2g2_sono_definiti_come .
?explanandum t3:hasReference ?comma .
}
UNION
{
?explanans t3:undergoer t3:2g2_sono_definiti_come .
?explanans t3:hasCondition ?condition .
}
}
