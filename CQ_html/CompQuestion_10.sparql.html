### 10) Come vengono definiti i terreni abbandonati caratterizzati da colture agricole?

SELECT ?comma ?explanandum ?exception ?explanans ?condition ?reference
WHERE {
{
?explanandum t3:actor t3:2g3_sono_definiti_come .
?explanandum t3:hasReference ?comma .
?explanandum t3:hasException ?exception .
}
UNION
{
?explanans t3:undergoer t3:2g3_sono_definiti_come .
?explanans t3:hasCondition ?condition .
?condition t3:hasMetaAttr ?reference .
}
}
