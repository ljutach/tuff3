### 8) Come vengono definiti i terreni abbandonati caratterizzati dal bosco ceduo?

SELECT ?comma ?explanandum ?explanans ?metric ?condition
WHERE {
{
?explanandum t3:actor t3:2g1_sono_definiti_come .
?explanandum t3:hasReference ?comma .
}
UNION
{
?explanans t3:undergoer t3:2g1_sono_definiti_come .
?explanans t3:hasMetric ?metric .
?metric t3:hasMetaAttr ?condition .
}
}
