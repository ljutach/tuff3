### 1) Quali termini vengono equiparati al termine bosco?

SELECT ?comma ?explanandum ?explanans
WHERE {
{
?explanandum t3:actor t3:semanticamente_equivalente_a .
?explanans t3:undergoer t3:semanticamente_equivalente_a .
}
UNION
{
?explanandum t3:actor t3:semanticamente_equivalente_a .
?explanandum t3:hasReference ?comma .
}
}
