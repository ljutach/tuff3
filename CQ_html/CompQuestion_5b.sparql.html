SELECT ?comma ?explanandum ?explanans ?function ?exception
WHERE {
{
?explanandum t3:actor t3:2d_si_definisce_come .
?explanandum t3:hasReference ?comma .
?explanans t3:undergoer t3:2d_si_definisce_come .

?explanans t3:hasException ?exception .
}
}
